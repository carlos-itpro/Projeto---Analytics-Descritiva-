OBJETIVO

Explorar os dados e identificar:

  • Ramos de Atividade predominantes por bairro.

  • Bairros com maior/menor incidncia de atividade econmica geradora de incmodo.

  • Bairros com maior oferta de servios proporcionalmente populao.

    Ex: Educao, Sade(Fitness), Lazer.

  • Servios direcionados ao gnero.

    Ex. Esttica(Feminino)

TRATAMENTO DOS DADOS

  • Excluir colunas: data_encerramento, situacao_empresa
  • Renomear colunas:
    • Total (habitantes_total)
    • Homens (habitantes_homens)
    • Mulheres (habitantes_mulheres)
  • Preencher dados faltantes em nome_fantasia com NENHUM
library(tidyverse)
database <- read_csv("database_full.csv",show_col_types = FALSE)

database <- select(database, - data_encerramento)
database <- select(database, - situacao_empresa)

database <- rename(database, c("habitantes_total" = "Total", 
                               "habitantes_homens" = "Homens",
                               "habitantes_mulheres" = "Mulheres"))

database$nome_fantasia[database$nome_fantasia == "No Existe"] <- "NENHUM"
database$nome_fantasia[database$nome_fantasia == "."] <- "NENHUM"

BAIRROS COM MAIOR OFERTA DE SERVIOS PROPORCIONALMENTE POPULAO

library(tidyverse)
database <- read_csv("database_full.csv",show_col_types = FALSE)

frequentes_2 <- database %>% 
  group_by(nome_bairro,nome_grupo,desc_atividade,Total) %>%
  count(sort = TRUE)

frequentes_3 <- database %>% 
  group_by(nome_bairro,nome_grupo,desc_atividade,Mulheres) %>%
  count(sort = TRUE)

fitness <- frequentes_2 %>%
  filter(desc_atividade == "ATIVIDADES DE CONDICIONAMENTO FSICO")

fitness <- fitness %>% 
  mutate(proporcao = Total/n)  %>% 
  arrange(proporcao)

educacao <- frequentes_2 %>%
  filter(desc_atividade == "EDUCAO INFANTIL - PR-ESCOLA")

educacao <- educacao %>% 
  mutate(proporcao = Total/n)  %>% 
  arrange(proporcao)

lazer <- frequentes_2 %>%
  filter(desc_atividade == "PRODUO MUSICAL")

lazer <- lazer %>% 
  mutate(proporcao = Total/n)  %>% 
  arrange(proporcao)
  
estetica <- frequentes_3 %>%
  filter(desc_atividade == "CABELEIREIROS, MANICURE E PEDICURE")

estetica <- estetica %>% 
  mutate(proporcao = Mulheres/n)  %>% 
  arrange(proporcao)
  

write_csv(fitness,"fitness.csv")
write_csv(educacao,"educacao.csv")
write_csv(lazer,"lazer.csv")
write_csv(estetica,"estetica.csv")
